How to Send Metrics (with Authentication)

Send metrics to this dashboard via POST request:

Endpoint: POST {{ base_url }}/api/metrics

Authentication: API Key required (contact admin)

Method 1: Include API Key in JSON Body

import requests
import psutil
from datetime import datetime

# YOUR API KEY (keep this secret!)
API_KEY = "your-api-key-here"

def send_metrics():
    metrics = {
        'api_key': API_KEY,  # <-- API key included in body
        'timestamp': datetime.now().isoformat(),
        'cpu_percent': psutil.cpu_percent(interval=1),
        'ram': {
            'used_gb': psutil.virtual_memory().used / (1024**3),
            'total_gb': psutil.virtual_memory().total / (1024**3),
            'percent': psutil.virtual_memory().percent
        },
        'client_name': 'My Computer',  # Optional identifier
        'gpu_percent': None,  # Optional: Add GPU monitoring
        'ping_ms': None,  # Optional: Add ping monitoring
        'internet_connected': True  # Optional: Internet status
    }
    
    try:
        response = requests.post(
            '{{ base_url }}/api/metrics',
            json=metrics,
            timeout=10
        )
        
        if response.status_code == 200:
            print(f"‚úì Metrics sent successfully: {response.json()}")
        else:
            print(f"‚úó Error: {response.status_code} - {response.json()}")
    
    except requests.exceptions.RequestException as e:
        print(f"‚úó Failed to send metrics: {e}")

# Run every 5 seconds
import time
while True:
    send_metrics()
    time.sleep(5)

Method 2: Include API Key in HTTP Header (More Secure)

import requests
import psutil
from datetime import datetime

# YOUR API KEY (keep this secret!)
API_KEY = "your-api-key-here"

def send_metrics():
    metrics = {
        'timestamp': datetime.now().isoformat(),
        'cpu_percent': psutil.cpu_percent(interval=1),
        'ram': {
            'used_gb': psutil.virtual_memory().used / (1024**3),
            'total_gb': psutil.virtual_memory().total / (1024**3),
            'percent': psutil.virtual_memory().percent
        },
        'client_name': 'My Computer',  # Optional identifier
        'gpu_percent': None,  # Optional: Add GPU monitoring
        'ping_ms': None,  # Optional: Add ping monitoring
        'internet_connected': True  # Optional: Internet status
    }
    
    # API key sent in header instead of body
    headers = {
        'X-API-Key': API_KEY,  # <-- API key in header
        'Content-Type': 'application/json'
    }
    
    try:
        response = requests.post(
            '{{ base_url }}/api/metrics',
            json=metrics,
            headers=headers,
            timeout=10
        )
        
        if response.status_code == 200:
            print(f"‚úì Metrics sent successfully: {response.json()}")
        elif response.status_code == 401:
            print(f"‚úó Authentication failed: No API key provided")
        elif response.status_code == 403:
            print(f"‚úó Authentication failed: Invalid API key")
        else:
            print(f"‚úó Error: {response.status_code} - {response.json()}")
    
    except requests.exceptions.RequestException as e:
        print(f"‚úó Failed to send metrics: {e}")

# Run every 5 seconds
import time
while True:
    send_metrics()
    time.sleep(5)

üîê Security Best Practices:

  • Never commit your API key to Git - Use environment variables
  • Use HTTPS - API keys should never be sent over HTTP
  • Rotate keys regularly - Change your API key periodically
  • Use header method - Sending keys in headers is more secure than in body
  • Keep keys secret - Don't share your API key with unauthorized users
